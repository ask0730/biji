# NOPSS文章抓取配置文件
# 配置文件说明：
# 1. 可以设置天数（days），表示获取最近N天的文章
# 2. 可以设置日期范围（start_date和end_date），表示获取指定日期范围内的文章
# 3. 如果都不设置，默认使用最近7天

# 方式1：设置天数（推荐）
# 例如：获取最近7天的文章
# days=7

# 方式2：设置日期范围
# 取消下面的注释并设置日期，格式：YYYY-MM-DD
 start_date=2023-04-05
 end_date=2026-04-10

# 方式3：设置Word文档输出路径
# 如果不设置，默认使用"项目申报文章"目录
 output_dir=D:\Desktop\输出
